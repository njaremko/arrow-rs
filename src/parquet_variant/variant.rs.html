34const MAX_SHORT_STRING_BYTES: usize = 0x3F;
35
36/// A Variant [`ShortString`]
37///
38/// This implementation is a zero cost wrapper over `&str` that ensures
39/// the length of the underlying string is a valid Variant short string (63 bytes or less)
40#[derive(Debug, Clone, Copy, PartialEq)]
41pub struct ShortString<'a>(pub(crate) &'a str);
42
43impl<'a> ShortString<'a> {
44    /// Attempts to interpret `value` as a variant short string value.  
45    ///
46    /// # Validation
47    ///
48    /// This constructor verifies that `value` is shorter than or equal to `MAX_SHORT_STRING_BYTES`
49    pub fn try_new(value: &'a str) -> Result<Self, ArrowError> {
50        if value.len() > MAX_SHORT_STRING_BYTES {
51            return Err(ArrowError::InvalidArgumentError(format!(
52                "value is larger than {MAX_SHORT_STRING_BYTES} bytes"
53            )));
54        }
55
56        Ok(Self(value))
57    }
58
59    /// Returns the underlying Variant short string as a &str
60    pub fn as_str(&self) -> &'a str {
61        self.0
62    }
63}
64
65impl<'a> From<ShortString<'a>> for &'a str {
66    fn from(value: ShortString<'a>) -> Self {
67        value.0
68    }
69}
70
71impl<'a> TryFrom<&'a str> for ShortString<'a> {
72    type Error = ArrowError;
73
74    fn try_from(value: &'a str) -> Result<Self, Self::Error> {
75        Self::try_new(value)
76    }
77}
78
79impl<'a> AsRef<str> for ShortString<'a> {
80    fn as_ref(&self) -> &str {
81        self.0
82    }
83}
84
85impl<'a> Deref for ShortString<'a> {
86    type Target = str;
87
88    fn deref(&self) -> &Self::Target {
89        self.0
90    }
91}

93/// Represents a [Parquet Variant]
94///
95/// The lifetimes `'m` and `'v` are for metadata and value buffers, respectively.
96///
97/// # Background
98///
99/// The [specification] says:
100///
101/// The Variant Binary Encoding allows representation of semi-structured data
102/// (e.g. JSON) in a form that can be efficiently queried by path. The design is
103/// intended to allow efficient access to nested data even in the presence of
104/// very wide or deep structures.
105///
106/// Another motivation for the representation is that (aside from metadata) each
107/// nested Variant value is contiguous and self-contained. For example, in a
108/// Variant containing an Array of Variant values, the representation of an
109/// inner Variant value, when paired with the metadata of the full variant, is
110/// itself a valid Variant.
111///
112/// When stored in Parquet files, Variant fields can also be *shredded*. Shredding
113/// refers to extracting some elements of the variant into separate columns for
114/// more efficient extraction/filter pushdown. The [Variant Shredding
115/// specification] describes the details of shredding Variant values as typed
116/// Parquet columns.
117///
118/// A Variant represents a type that contains one of:
119///
120/// * Primitive: A type and corresponding value (e.g. INT, STRING)
121///
122/// * Array: An ordered list of Variant values
123///
124/// * Object: An unordered collection of string/Variant pairs (i.e. key/value
125///   pairs). An object may not contain duplicate keys.
126///
127/// # Encoding
128///
129/// A Variant is encoded with 2 binary values, the value and the metadata. The
130/// metadata stores a header and an optional dictionary of field names which are
131/// referred to by offset in the value. The value is a binary representation of
132/// the actual data, and varies depending on the type.
133///
134/// # Design Goals
135///
136/// The design goals of the Rust API are as follows:
137/// 1. Speed / Zero copy access (no `clone`ing is required)
138/// 2. Safety
139/// 3. Follow standard Rust conventions
140///
141/// [Parquet Variant]: https://github.com/apache/parquet-format/blob/master/VariantEncoding.md
142/// [specification]: https://github.com/apache/parquet-format/blob/master/VariantEncoding.md
143/// [Variant Shredding specification]: https://github.com/apache/parquet-format/blob/master/VariantShredding.md
144///
145/// # Examples:
146///
147/// ## Creating `Variant` from Rust Types
148/// ```
149/// use parquet_variant::Variant;
150/// // variants can be directly constructed
151/// let variant = Variant::Int32(123);
152/// // or constructed via `From` impls
153/// assert_eq!(variant, Variant::from(123i32));
154/// ```
155/// ## Creating `Variant` from metadata and value
156/// ```
157/// # use parquet_variant::{Variant, VariantMetadata};
158/// let metadata = [0x01, 0x00, 0x00];
159/// let value = [0x09, 0x48, 0x49];
160/// // parse the header metadata
161/// assert_eq!(
162///   Variant::from("HI"),
163///   Variant::try_new(&metadata, &value).unwrap()
164/// );
165/// ```
166///
167/// ## Using `Variant` values
168/// ```
169/// # use parquet_variant::Variant;
170/// # let variant = Variant::Int32(123);
171/// // variants can be used in match statements like normal enums
172/// match variant {
173///   Variant::Int32(i) => println!("Integer: {}", i),
174///   Variant::String(s) => println!("String: {}", s),
175///   _ => println!("Other variant"),
176/// }
177/// ```
178#[derive(Clone, Debug, PartialEq)]
179pub enum Variant<'m, 'v> {
180    /// Primitive type: Null
181    Null,
182    /// Primitive (type_id=1): INT(8, SIGNED)
183    Int8(i8),
184    /// Primitive (type_id=1): INT(16, SIGNED)
185    Int16(i16),
186    /// Primitive (type_id=1): INT(32, SIGNED)
187    Int32(i32),
188    /// Primitive (type_id=1): INT(64, SIGNED)
189    Int64(i64),
190    /// Primitive (type_id=1): DATE
191    Date(NaiveDate),
192    /// Primitive (type_id=1): TIMESTAMP(isAdjustedToUTC=true, MICROS)
193    TimestampMicros(DateTime<Utc>),
194    /// Primitive (type_id=1): TIMESTAMP(isAdjustedToUTC=false, MICROS)
195    TimestampNtzMicros(NaiveDateTime),
196    /// Primitive (type_id=1): DECIMAL(precision, scale) 32-bits
197    Decimal4 { integer: i32, scale: u8 },
198    /// Primitive (type_id=1): DECIMAL(precision, scale) 64-bits
199    Decimal8 { integer: i64, scale: u8 },
200    /// Primitive (type_id=1): DECIMAL(precision, scale) 128-bits
201    Decimal16 { integer: i128, scale: u8 },
202    /// Primitive (type_id=1): FLOAT
203    Float(f32),
204    /// Primitive (type_id=1): DOUBLE
205    Double(f64),
206    /// Primitive (type_id=1): BOOLEAN (true)
207    BooleanTrue,
208    /// Primitive (type_id=1): BOOLEAN (false)
209    BooleanFalse,
210    // Note: only need the *value* buffer for these types
211    /// Primitive (type_id=1): BINARY
212    Binary(&'v [u8]),
213    /// Primitive (type_id=1): STRING
214    String(&'v str),
215    /// Short String (type_id=2): STRING
216    ShortString(ShortString<'v>),
217    // need both metadata & value
218    /// Object (type_id=3): N/A
219    Object(VariantObject<'m, 'v>),
220    /// Array (type_id=4): N/A
221    List(VariantList<'m, 'v>),
222}

224impl<'m, 'v> Variant<'m, 'v> {
225    /// Create a new `Variant` from metadata and value.
226    ///
227    /// # Example
228    /// ```
229    /// use parquet_variant::{Variant, VariantMetadata};
230    /// let metadata = [0x01, 0x00, 0x00];
231    /// let value = [0x09, 0x48, 0x49];
232    /// // parse the header metadata
233    /// assert_eq!(
234    ///   Variant::from("HI"),
235    ///   Variant::try_new(&metadata, &value).unwrap()
236    /// );
237    /// ```
238    pub fn try_new(metadata: &'m [u8], value: &'v [u8]) -> Result<Self, ArrowError> {
239        let metadata = VariantMetadata::try_new(metadata)?;
240        Self::try_new_with_metadata(metadata, value)
241    }
242
243    /// Create a new variant with existing metadata
244    ///
245    /// # Example
246    /// ```
247    /// # use parquet_variant::{Variant, VariantMetadata};
248    /// let metadata = [0x01, 0x00, 0x00];
249    /// let value = [0x09, 0x48, 0x49];
250    /// // parse the header metadata first
251    /// let metadata = VariantMetadata::try_new(&metadata).unwrap();
252    /// assert_eq!(
253    ///   Variant::from("HI"),
254    ///   Variant::try_new_with_metadata(metadata, &value).unwrap()
255    /// );
256    /// ```
257    pub fn try_new_with_metadata(
258        metadata: VariantMetadata<'m>,
259        value: &'v [u8],
260    ) -> Result<Self, ArrowError> {
261        let value_metadata = first_byte_from_slice(value)?;
262        let value_data = slice_from_slice(value, 1..)?;
263        let new_self = match get_basic_type(value_metadata) {
264            VariantBasicType::Primitive => match get_primitive_type(value_metadata)? {
265                VariantPrimitiveType::Null => Variant::Null,
266                VariantPrimitiveType::Int8 => Variant::Int8(decoder::decode_int8(value_data)?),
267                VariantPrimitiveType::Int16 => Variant::Int16(decoder::decode_int16(value_data)?),
268                VariantPrimitiveType::Int32 => Variant::Int32(decoder::decode_int32(value_data)?),
269                VariantPrimitiveType::Int64 => Variant::Int64(decoder::decode_int64(value_data)?),
270                VariantPrimitiveType::Decimal4 => {
271                    let (integer, scale) = decoder::decode_decimal4(value_data)?;
272                    Variant::Decimal4 { integer, scale }
273                }
274                VariantPrimitiveType::Decimal8 => {
275                    let (integer, scale) = decoder::decode_decimal8(value_data)?;
276                    Variant::Decimal8 { integer, scale }
277                }
278                VariantPrimitiveType::Decimal16 => {
279                    let (integer, scale) = decoder::decode_decimal16(value_data)?;
280                    Variant::Decimal16 { integer, scale }
281                }
282                VariantPrimitiveType::Float => Variant::Float(decoder::decode_float(value_data)?),
283                VariantPrimitiveType::Double => {
284                    Variant::Double(decoder::decode_double(value_data)?)
285                }
286                VariantPrimitiveType::BooleanTrue => Variant::BooleanTrue,
287                VariantPrimitiveType::BooleanFalse => Variant::BooleanFalse,
288                VariantPrimitiveType::Date => Variant::Date(decoder::decode_date(value_data)?),
289                VariantPrimitiveType::TimestampMicros => {
290                    Variant::TimestampMicros(decoder::decode_timestamp_micros(value_data)?)
291                }
292                VariantPrimitiveType::TimestampNtzMicros => {
293                    Variant::TimestampNtzMicros(decoder::decode_timestampntz_micros(value_data)?)
294                }
295                VariantPrimitiveType::Binary => {
296                    Variant::Binary(decoder::decode_binary(value_data)?)
297                }
298                VariantPrimitiveType::String => {
299                    Variant::String(decoder::decode_long_string(value_data)?)
300                }
301            },
302            VariantBasicType::ShortString => {
303                Variant::ShortString(decoder::decode_short_string(value_metadata, value_data)?)
304            }
305            VariantBasicType::Object => Variant::Object(VariantObject::try_new(metadata, value)?),
306            VariantBasicType::Array => Variant::List(VariantList::try_new(metadata, value)?),
307        };
308        Ok(new_self)
309    }

311    /// Converts this variant to `()` if it is null.
312    ///
313    /// Returns `Some(())` for null variants,
314    /// `None` for non-null variants.
315    ///
316    /// # Examples
317    ///
318    /// ```
319    /// use parquet_variant::Variant;
320    ///
321    /// // you can extract `()` from a null variant
322    /// let v1 = Variant::from(());
323    /// assert_eq!(v1.as_null(), Some(()));
324    ///
325    /// // but not from other variants
326    /// let v2 = Variant::from("hello!");
327    /// assert_eq!(v2.as_null(), None);
328    /// ```
329    pub fn as_null(&self) -> Option<()> {
330        matches!(self, Variant::Null).then_some(())
331    }

338    /// # Examples

391    /// Converts this variant to a `DateTime<Utc>` if possible.
392    ///
393    /// Returns `Some(DateTime<Utc>)` for timestamp variants,
394    /// `None` for non-timestamp variants.
395    ///
396    /// # Examples
397    ///
398    /// ```
399    /// use parquet_variant::Variant;
400    /// use chrono::NaiveDate;
401    ///
402    /// // you can extract a DateTime<Utc> from a UTC-adjusted variant
403    /// let datetime = NaiveDate::from_ymd_opt(2025, 4, 16).unwrap().and_hms_milli_opt(12, 34, 56, 780).unwrap().and_utc();
404    /// let v1 = Variant::from(datetime);
405    /// assert_eq!(v1.as_datetime_utc(), Some(datetime));
406    ///
407    /// // or a non-UTC-adjusted variant
408    /// let datetime = NaiveDate::from_ymd_opt(2025, 4, 16).unwrap().and_hms_milli_opt(12, 34, 56, 780).unwrap();
409    /// let v2 = Variant::from(datetime);
410    /// assert_eq!(v2.as_datetime_utc(), Some(datetime.and_utc()));
411    ///
412    /// // but not from other variants
413    /// let v3 = Variant::from("hello!");
414    /// assert_eq!(v3.as_datetime_utc(), None);
415    /// ```
416    pub fn as_datetime_utc(&self) -> Option<DateTime<Utc>> {
417        match *self {
418            Variant::TimestampMicros(d) => Some(d),
419            Variant::TimestampNtzMicros(d) => Some(d.and_utc()),
420            _ => None,
421        }
422    }

634    /// Converts this variant to tuple with a 4-byte unscaled value if possible.
635    ///
636    /// Returns `Some((i32, u8))` for decimal variants where the unscaled value
637    /// fits in `i32` range,
638    /// `None` for non-decimal variants or decimal values that would overflow.
639    ///
640    /// # Examples
641    ///
642    /// ```
643    /// use parquet_variant::Variant;
644    ///
645    /// // you can extract decimal parts from smaller or equally-sized decimal variants
646    /// let v1 = Variant::from((1234_i32, 2));
647    /// assert_eq!(v1.as_decimal_int32(), Some((1234_i32, 2)));
648    ///
649    /// // and from larger decimal variants if they fit
650    /// let v2 = Variant::from((1234_i64, 2));
651    /// assert_eq!(v2.as_decimal_int32(), Some((1234_i32, 2)));
652    ///
653    /// // but not if the value would overflow i32
654    /// let v3 = Variant::from((12345678901i64, 2));
655    /// assert_eq!(v3.as_decimal_int32(), None);
656    ///
657    /// // or if the variant is not a decimal
658    /// let v4 = Variant::from("hello!");
659    /// assert_eq!(v4.as_decimal_int32(), None);
660    /// ```
661    pub fn as_decimal_int32(&self) -> Option<(i32, u8)> {
662        match *self {
663            Variant::Decimal4 { integer, scale } => Some((integer, scale)),
664            Variant::Decimal8 { integer, scale } => {
665                if let Ok(converted_integer) = integer.try_into() {
666                    Some((converted_integer, scale))
667                } else {
668                    None
669                }
670            }
671            Variant::Decimal16 { integer, scale } => {
672                if let Ok(converted_integer) = integer.try_into() {
673                    Some((converted_integer, scale))
674                } else {
675                    None
676                }
677            }
678            _ => None,
679        }
680    }

812    /// Converts this variant to an `Object` if it is an [`VariantObject`].
813    ///
814    /// Returns `Some(&VariantObject)` for object variants,
815    /// `None` for non-object variants.
816    ///
817    /// # Examples
818    /// ```
819    /// # use parquet_variant::{Variant, VariantBuilder, VariantObject};
820    /// # let (metadata, value) = {
821    /// # let mut builder = VariantBuilder::new();
822    /// #   let mut obj = builder.new_object();
823    /// #   obj.append_value("name", "John");
824    /// #   obj.finish();
825    /// #   builder.finish()
826    /// # };
827    /// // object that is {"name": "John"}
828    ///  let variant = Variant::try_new(&metadata, &value).unwrap();
829    /// // use the `as_object` method to access the object
830    /// let obj = variant.as_object().expect("variant should be an object");
831    /// assert_eq!(obj.field_by_name("name").unwrap(), Some(Variant::from("John")));
832    /// ```
833    pub fn as_object(&'m self) -> Option<&'m VariantObject<'m, 'v>> {
834        if let Variant::Object(obj) = self {
835            Some(obj)
836        } else {
837            None
838        }
839    }

842    ///
843    /// Returns `Some(&VariantList)` for list variants,
844    /// `None` for non-list variants.
845    ///
846    /// # Examples
847    /// ```
848    /// # use parquet_variant::{Variant, VariantBuilder, VariantList};
849    /// # let (metadata, value) = {

854    /// #   list.finish();
855    /// #   builder.finish()
856    /// # };
857    /// // list that is ["John", "Doe"]
858    /// let variant = Variant::try_new(&metadata, &value).unwrap();
859    /// // use the `as_list` method to access the list
860    /// let list = variant.as_list().expect("variant should be a list");
861    /// assert_eq!(list.len(), 2);

863    /// assert_eq!(list.get(1).unwrap(), Variant::from("Doe"));
864    /// ```
865    pub fn as_list(&'m self) -> Option<&'m VariantList<'m, 'v>> {
866        if let Variant::List(list) = self {
867            Some(list)
868        } else {
869            None
870        }
871    }
